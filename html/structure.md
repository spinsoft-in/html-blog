```html
  <!DOCTYPE html>
  <html>
    <head>
        <title> Page Title </title>
         <meta charset="UTF-8">
         <meta name="description" content="Learn2Build">
         <meta name="keywords" content="HTML, CSS, JavaScript">
         <meta name="author" content="Naresh Kumar">

         <link rel="stylesheet" href="styles.css">

        <script>
            // JavaScript
        </script>

            /* CSS */
        </style>

    <body>

          <nav>

          </nav>

        <aside>

        </aside>

          <div >
            <p>Page Content</p>
          </div>
        </section>

        <article>
          <h1>News </h1>
          <p> Latest news ... </p>
        </article>

      </main>
      <footer>

      </footer>
    </body>
```
